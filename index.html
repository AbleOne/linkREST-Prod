import org.glassfish.jersey.server.ResourceConfig;

                    @ApplicationPath("/")
                    public class MyApp extends ResourceConfig {

                        public MyApp() {

                            // Start Cayenne
                            ServerRuntime cayenneRuntime  = new ServerRuntime("my-cayenne.xml");

                            // Start LinkRest with default settings.
                            // (check out LinkRestBuilder API for more configuration options)
                            register(LinkRestBuilder.build(cayenneRuntime));
                        }
                    }
